1) Получить платеж
curl -X GET 'http://localhost:8080/api/payments/123e4567-e89b-12d3-a456-426614174000'

2) Создать платеж
curl -X POST 'http://localhost:8080/api/payments' \
  -H 'Content-Type: application/json' \
  -d '{
    "inquiryRefId": "123e4567-e89b-12d3-a456-426614174001",
    "amount": 200.00,
    "currency": "EUR",
    "transactionRefId": "123e4567-e89b-12d3-a456-426614174003",
    "status": "PENDING",
    "note": "Created"
  }'

3) Обновить платеж
curl -X PUT 'http://localhost:8080/api/payments/1eddc5f4-9e3f-43c7-b2cf-d976864d6953' \
  -H 'Content-Type: application/json' \
  -d '{
    "inquiryRefId": "123e4567-e89b-12d3-a456-426614174005",
    "amount": 300.00,
    "currency": "RUB",
    "transactionRefId": "123e4567-e89b-12d3-a456-426614174006",
    "status": "RECEIVED",
    "note": "Updated"
  }'

4) Обновить заметку платежа
curl -X PATCH 'http://localhost:8080/api/payments/1eddc5f4-9e3f-43c7-b2cf-d976864d6953/note' \
  -H 'Content-Type: application/json' \
  -d '{"note": "Платёж подтверждён. Ожидается зачисление."}'

5) Удалить платеж
curl -X DELETE 'http://localhost:8080/api/payments/1eddc5f4-9e3f-43c7-b2cf-d976864d6953'
